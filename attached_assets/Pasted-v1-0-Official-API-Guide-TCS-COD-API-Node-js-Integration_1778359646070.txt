Step 1: Bearer Token (Authorization)
Sab se pehle Bearer Token lena zaroori hai. Yeh token E-COM Authentication API mein use hota hai.

Field	Value
Method	GET
Endpoint	/auth/api/auth
Params	clientid, clientsecret — TCS se milenge
Returns	accessToken, expiry
 src/auth.js — getBearerToken()
javascript
const axios = require('axios');
const { BASE_URL, CLIENT_ID, CLIENT_SECRET } = require('./config');

// Token cache — baar baar request mat karo
let bearerToken = null;
let bearerExpiry = null;

async function getBearerToken() {
  // Agar token abhi valid hai toh naya mat lo
  if (bearerToken && bearerExpiry && new Date() < new Date(bearerExpiry)) {
    return bearerToken;
  }

  try {
    const response = await axios.get(`${BASE_URL}/auth/api/auth`, {
      data: {
        clientid: CLIENT_ID,
        clientsecret: CLIENT_SECRET,
      },
    });

    if (response.data.status && response.data.result) {
      bearerToken = response.data.result.accessToken;
      bearerExpiry = response.data.result.expiry;
      return bearerToken;
    }

    throw new Error('Bearer token lene mein masla: ' + JSON.stringify(response.data));
  } catch (err) {
    throw new Error('Authorization API error: ' + err.message);
  }
}

module.exports = { getBearerToken };

Step 2: E-COM Access Token
Bearer Token milne ke baad yeh E-COM token lein. Yeh accessToken har API ke JSON body mein dena hoga.

 src/auth.js — getAccessToken()
javascript
let accessToken = null;
let accessExpiry = null;

async function getAccessToken() {
  if (accessToken && accessExpiry && new Date() < new Date(accessExpiry)) {
    return accessToken;
  }

  const bearer = await getBearerToken();

  const response = await axios.get(
    `${BASE_URL}/ecom/api/authentication/token`,
    {
      data: {
        username: USERNAME,
        password: PASSWORD,
      },
      headers: {
        Authorization: `Bearer ${bearer}`,
        'Content-Type': 'application/json',
      },
    }
  );

  if (response.data.message === 'success') {
    accessToken = response.data.accesstoken;
    accessExpiry = response.data.expiry;
    return accessToken;
  }

  throw new Error('E-COM token error');
}

module.exports = { getBearerToken, getAccessToken };

 src/booking.js — createBooking()
javascript
const axios = require('axios');
const { BASE_URL, ACCOUNT } = require('./config');
const { getAccessToken, getBearerToken } = require('./auth');

async function createBooking(bookingData) {
  const accesstoken = await getAccessToken();
  const bearer = await getBearerToken();

  const payload = {
    accesstoken,
    shipperinfo: {
      tcsaccount: ACCOUNT,
      shippername: bookingData.shipperName,
      address1: bookingData.shipperAddress,
      countrycode: 'PK',
      countryname: 'Pakistan',
      citycode: bookingData.shipperCityCode,
      cityname: bookingData.shipperCity,
      mobile: bookingData.shipperMobile,
    },
    consigneeinfo: {
      firstname: bookingData.consigneeFirstName,
      middlename: bookingData.consigneeMiddleName || '.',
      lastname: bookingData.consigneeLastName || '',
      address1: bookingData.consigneeAddress,
      countrycode: 'PK',
      countryname: 'Pakistan',
      citycode: bookingData.consigneeCityCode,
      cityname: bookingData.consigneeCity,
      mobile: bookingData.consigneeMobile,
      email: bookingData.consigneeEmail || '',
    },
    shipmentinfo: {
      costcentercode: bookingData.costCenterCode,
      referenceno: bookingData.orderRef || '',
      servicecode: bookingData.serviceCode || 'O',
      currency: 'PKR',
      codamount: bookingData.codAmount,
      weightinkg: bookingData.weight,
      pieces: bookingData.pieces || 1,
      fragile: bookingData.fragile || false,
      remarks: bookingData.remarks || '',
      skus: [
        {
          description: bookingData.itemDesc || 'Product',
          quantity: bookingData.pieces || 1,
          weight: bookingData.weight,
          unitprice: bookingData.codAmount,
        },
      ],
    },
  };

  const res = await axios.post(
    `${BASE_URL}/ecom/api/booking/create`,
    payload,
    { headers: { Authorization: `Bearer ${bearer}` } }
  );

  return res.data; // { response, consignmentNo, code, status, message }
}

module.exports = { createBooking };

 src/tracking.js — trackShipment()
javascript
const axios = require('axios');
const { BASE_URL } = require('./config');
const { getAccessToken, getBearerToken } = require('./auth');

/**
 * @param {string|string[]} consignmentNumbers - Ek ya zyada CN numbers
 */
async function trackShipment(consignmentNumbers) {
  const bearer = await getBearerToken();
  const cns = Array.isArray(consignmentNumbers)
    ? consignmentNumbers
    : [consignmentNumbers];

  const res = await axios.get(
    `${BASE_URL}/tracking/api/Tracking/GetDynamicTrackDetail`,
    {
      data: { consignee: cns },
      headers: { Authorization: `Bearer ${bearer}` },
    }
  );

  // Returns: shipmentinfo, deliveryinfo, checkpoints, shipmentsummary
  return res.data;
}

module.exports = { trackShipment };

Error Handling — Best Practices
Production-ready error handling wrapper jo sab APIs mein use karna chahiye.

Code	Meaning	Kya Karen
200	OK — Kaam ho gaya	Response data use karen
201	Created	Naya record ban gaya
400	Bad Request	Request fields check karen
401	Auth Failure	Token dobara lein ya credentials check karen
403	Forbidden	Permission nahi hai
404	Not Found	CN ya resource maujood nahi
500	Server Error	TCS ki taraf se masla, retry karen
503	Service Down	Kuch der baad try karen
 src/utils.js — apiCall() wrapper
javascript
/**
 * Production-ready API call wrapper
 * Automatic retry on 401 (token refresh) aur 503 (service down)
 */
async function apiCall(fn, maxRetries = 3) {
  let lastError;

  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      const result = await fn();
      return { success: true, data: result };

    } catch (err) {
      lastError = err;
      const status = err.response?.status;

      if (status === 401) {
        // Token expired — reset aur retry
        console.log('Token expired, naya le raha hoon...');
        bearerToken = null; accessToken = null;
        if (attempt < maxRetries) continue;
      }

      if (status === 503 && attempt < maxRetries) {
        // Service temporarily down — wait aur retry
        await new Promise(r => setTimeout(r, 2000 * attempt));
        continue;
      }

      // Baaki errors ke liye turant fail
      break;
    }
  }

  return {
    success: false,
    error: lastError.response?.data || lastError.message,
    status: lastError.response?.status,
  };
}

require('dotenv').config();
const { createBooking, cancelBooking, printLabel } = require('./src/booking');
const { trackShipment } = require('./src/tracking');
const { getPaymentStatus } = require('./src/payment');

async function main() {
  try {
    console.log('=== TCS API Integration Test ===');

    // 1. BOOKING BANAO
    console.log('\n📦 Booking create kar raha hoon...');
    const booking = await createBooking({
      shipperName: 'My Online Store',
      shipperAddress: 'Shop 5, Liberty Market',
      shipperCityCode: 'LHE',
      shipperCity: 'Lahore',
      shipperMobile: '03001234567',

      consigneeFirstName: 'Ahmed',
      consigneeMiddleName: 'Ali',
      consigneeLastName: 'Khan',
      consigneeAddress: 'House 10, Block B, DHA Phase 5',
      consigneeCityCode: 'KHI',
      consigneeCity: 'Karachi',
      consigneeMobile: '03219876543',
      consigneeEmail: 'ahmed@email.com',

      costCenterCode: 'Test-01',
      orderRef: 'ORDER-12345',
      serviceCode: 'O',
      codAmount: 2500,
      weight: 1.5,
      pieces: 1,
      fragile: false,
      itemDesc: 'Kapray',
      remarks: 'Handle with care',
    });

    if (!booking.status) {
      throw new Error('Booking fail: ' + booking.message);
    }

    const cn = booking.consignmentNo;
    console.log(`✅ Booking SUCCESS! CN: ${cn}`);

    // 2. LABEL PRINT
    console.log('\n🖨️  Label print kar raha hoon...');
    const label = await printLabel(cn);
    console.log('Label response:', label);

    // 3. TRACKING
    console.log('\n📍 Tracking check kar raha hoon...');
    const tracking = await trackShipment(cn);
    if (tracking.shipmentsummary) {
      console.log('Status:', tracking.shipmentsummary);
    }

    // 4. PAYMENT STATUS
    console.log('\n💳 Payment status check...');
    const payment = await getPaymentStatus('YOUR_CUSTOMER_NO', cn);
    console.log('Payment:', payment.detail?.[0]);

  } catch (err) {
    console.error('❌ Error:', err.message);
    if (err.response) {
      console.error('Response data:', err.response.data);
    }
    process.exit(1);
  }
}

main();
